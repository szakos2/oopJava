
public class asd
{
	public static void main (String[] args)
	{
		// 5) Órai feladat
		System.out.println("\n\nÖtödik feladat:");
		Kiir ();
	}
	
	static void Kiir ()
	{
		for (int i = 1; i <= 5; i++)
		{
			for (int j = 1; j <= i; j++)
			{
				System.out.print(i);
			}
			System.out.println("");
		}
	}
}
