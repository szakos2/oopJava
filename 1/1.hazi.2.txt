
public class asd
{
	public static void main (String[] args)
	{
		// 2) Házi feladat
		System.out.println("\nMasodik házi feladat:");
		PitagorasziSzamharmasok ();
	}
	
	static void PitagorasziSzamharmasok ()
	{
		for (int a = 1; a <= 50; a++)
		{
			for (int b = 1; b <= 50; b++)
			{
				for (int c = 1; c <= 50; c++)
				{
					if (a * a + b * b == c * c && a < b && b < c)
					{
						System.out.println(a + " " + b +" " + c);
					}
				}
			}
		}
	}
}
