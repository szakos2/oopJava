
public class asd
{
	public static void main (String[] args)
	{
		// 2) Órai feladat
		System.out.println("\nMasodik feladat:");
		ElsoTizFaktorialis ();
	}
	
	static void ElsoTizFaktorialis ()
	{
		for (int i = 1; i <= 10; i++)
		{
			int faktorialis = 1;
			for (int j = 1; j <= i; j++)
			{
				faktorialis *= j;
			}
			System.out.println(i + ": " +faktorialis);
		}
	}
}
